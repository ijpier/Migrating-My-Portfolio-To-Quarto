all_nba <- function(x){
  team_inside <- read_excel("data/NBA Team Total Data 2024-2025.xlsx", sheet = x)
  team_inside <- team_inside %>%
  mutate(
    Team = x,
    Won_award = ifelse(is.na(Awards), "0", "1"),
    PRA=PTS+TRB+AST,
    STOCKS=STL+BLK) 
  print(team_inside)
}
